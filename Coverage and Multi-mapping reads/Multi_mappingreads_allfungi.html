Shorten chromosome names, divide into subgenomes and arrange

A. rouxii

# Shorten the chromosome name so it only includes chromosome number
# Add column indicating subgenome
AmyRoux_reads_per_chrom_clean <- AmyRoux_reads_per_chrom_filtered %>% 
  # Create subgenome column. Classify based on the part before the first "_". 
  mutate(subgenome = str_extract(contig, "^[^_]+")) %>%
  # Remove everything before the last "_", only keeping the chromosome number
  mutate(contig_short = str_extract(contig, "[^_]+$")) %>% 
  mutate(subgenome = str_replace(subgenome, "subgenom(\\d+)", "subgenome\\1"))
# Arrange chromosomes for plotting
AmyRoux_reads_per_chrom_clean <- AmyRoux_reads_per_chrom_clean %>%
  # sort by subgenome then chromosome
  arrange(subgenome, as.numeric(contig_short)) %>%   
  # unique for factor
  mutate(
    contig_unique = paste(subgenome, contig_short, sep = "_"),  
    contig_unique = factor(contig_unique, levels = contig_unique)
  )

Plotting

# Set upper limit for plots so that the y-axis is the same for all species
AmyRoux_Mb_plot <- ggplot(AmyRoux_reads_per_chrom_clean, 
                       aes(x = contig_unique, y = mm_reads_per_Mb, fill = subgenome)) +
  geom_bar(stat = "identity") +
  theme_bw() +
  theme(axis.text.x = element_text(angle = 90), 
        legend.position = "none") +
  labs(title = "A.rouxii", x = "Chromosome", y = "Multi-mapped reads/Mb") +
  scale_x_discrete(labels = AmyRoux_reads_per_chrom_clean$contig_short) +
